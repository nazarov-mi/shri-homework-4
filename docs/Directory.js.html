const {
	REPOSITORY_PATH
} = require('./config')

const List = require('./List')
const DirectoryObject = require('./DirectoryObject')

/**
 * Класс — представление директории Git
 * @class
 * @extends List
 */
class Directory extends List {

	/**
	 * Создаёт экземпляр Directory
	 * @constructs
	 */
	constructor () {
		super()

		this.history = []
	}

	clearHistory () {
		this.history = []
	}

	getLastPoint () {
		const history = this.history
		const len = history.length

		return len > 0 ? history[len - 1] : null
	}

	back () {
		this.history.pop()
	}

	/**
	 * @inheritdoc
	 */
	change (hash) {
		const promise = super.change(hash)
		const lastPoint = this.getLastPoint()

		if (lastPoint && lastPoint.hash === this._hash) {
			this.history.pop()
		} else {
			this.history.push({
				hash: this._hash
			})
		}

		return promise
	}

	/**
	 * @inheritdoc
	 */
	async fetch (hash) {
		const res = await this._reader.start('git', ['ls-tree', '--full-name', hash], {
			cwd: REPOSITORY_PATH
		})

		const data = this.constructor._parse(res)

		this._setData(data)
	}

	/**
	 * Парсит строки в объекты DirectoryObject
	 * @private
	 * @param  {String} data - Строки
	 * @return {Array<DirectoryObject>}
	 */
	static _parse (data) {
		const rows = data.split(/\n/).filter(row => (!!row))

		return rows.map((row) => {
			const obj = new DirectoryObject()

			obj.parse(row)

			return obj
		})
	}
}

module.exports = Directory
